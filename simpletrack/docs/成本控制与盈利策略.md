```go
// 根据用户计划调整频率
type AISchedule struct {
    Free  time.Duration // 不提供AI功能
    Pro   time.Duration // 每周1次

func (s *AIService) ShouldGenerateInsight(siteID int, plan string) bool {
    lastInsight, _ := s.repo.GetLastInsight(siteID)

    case "pro":
        return time.Since(lastInsight.CreatedAt) >= 7*24*time.Hour
    default:
        return false
    }
}
```

```go
func (s *AIService) GenerateInsightWithCache(ctx context.Context, siteID int) (*models.Insight, error) {
    // 1. 检查是否有相似的数据模式
    cacheKey := s.generateCacheKey(siteID)
    if cached, err := s.cache.Get(ctx, cacheKey); err == nil {
        return cached, nil

    // 2. 生成新的洞察
    insight, err := s.generateInsight(ctx, siteID)
    if err != nil {
        return nil, err
    }

    // 3. 缓存结果（7天）
    s.cache.Set(ctx, cacheKey, insight, 7*24*time.Hour)

    return insight, nil
}
```

```go
func (h *Handler) GetStats(c *gin.Context) {

        c.JSON(200, cached)

    stats, _ := h.service.GetStats(siteID)

    h.cache.Set(cacheKey, stats, 5*time.Minute)

}
```

### 限制维度设计

#### 1. 事件数量限制（核心）

```go
// 免费版：1000事件/月
// 付费版：100,000事件/月

type RateLimiter struct {
    redis *redis.Client
}

func (r *RateLimiter) CheckEventLimit(ctx context.Context, siteID int, plan string) error {
    // 获取本月已使用事件数
    key := fmt.Sprintf("events:%d:%s", siteID, time.Now().Format("2006-01"))
    count, err := r.redis.Get(ctx, key).Int()
    if err != nil && err != redis.Nil {
        return err
    }

    // 检查限制
    limit := r.getLimit(plan)
    if count >= limit {
        return ErrRateLimitExceeded
    }

    // 增加计数
    r.redis.Incr(ctx, key)
    r.redis.Expire(ctx, key, 31*24*time.Hour)

    return nil
}

func (r *RateLimiter) getLimit(plan string) int {
    switch plan {
    case "free":
        return 1000
    case "pro":
        return 100000
    default:
        return 0
    }
}
```

**为什么1000事件？**
- ✅ 足够体验产品（小网站1周流量）
- ✅ 成本可控（存储<1MB）
- ✅ 引导升级（很容易超出）

```go
    return func(c *gin.Context) {
        userID := c.GetInt("user_id")

        // 获取限制
        limit := m.getAPILimit(plan)
```

```go
// 达到限制后立即停止
if count >= limit {
    return ErrRateLimitExceeded
}
```

**优势**：
- ✅ 成本完全可控
- ✅ 强烈的升级动机

**劣势**：
- ❌ 用户体验可能不好

---

#### 软限制（不推荐）

```go
// 达到限制后继续收集，但提示升级
if count >= limit {
    // 发送提醒邮件
    s.sendUpgradeReminder(userID)
    // 但继续收集数据
}
```

**优势**：
- ✅ 用户体验好

**劣势**：
- ❌ 成本可能失控
- ❌ 用户没有升级动力

**建议**：使用硬限制

```go
    // 获取用户行为数据

    eventCount, _ := p.repo.GetRecentEventCount(ctx, userID, 7*24*time.Hour)
```

```go
        count, _ := m.redis.Incr(c.Request.Context(), key).Result()
        m.redis.Expire(c.Request.Context(), key, time.Minute)

        if count > 100 { // 每分钟最多100个事件
            c.JSON(429, gin.H{"error": "Too many requests"})
            c.Abort()
            return
        }

        c.Next()
    }
}

// 2. 用户级别限流
func (r *RateLimiter) CheckUserLimit(ctx context.Context, siteID int) error {
    key := fmt.Sprintf("user_limit:%d:%d", siteID, time.Now().Unix()/60)
    count, _ := r.redis.Incr(ctx, key).Result()
    r.redis.Expire(ctx, key, time.Minute)

    if count > 1000 { // 每分钟最多1000个事件
        return ErrTooManyRequests
    }

    return nil
}

// 3. 异常检测
func (s *AnomalyDetector) DetectSpam(ctx context.Context, siteID int) bool {
    // 获取最近1分钟的事件数
    count, _ := s.repo.GetRecentEventCount(ctx, siteID, time.Minute)

    // 获取历史平均值
    avg, _ := s.repo.GetAverageEventCount(ctx, siteID)

    // 如果超过平均值10倍，可能是spam
    if count > avg*10 {
        // 自动暂停该网站
        s.repo.SuspendSite(ctx, siteID)

        // 通知管理员
        s.notifyAdmin(siteID, count, avg)

        return true
    }

    return false
}
```

```go
// 管理员仪表盘
type CostDashboard struct {
    ServerCost    float64 `json:"server_cost"`
    DatabaseCost  float64 `json:"database_cost"`
    AICost        float64 `json:"ai_cost"`
    EmailCost     float64 `json:"email_cost"`
    TotalCost     float64 `json:"total_cost"`
    Revenue       float64 `json:"revenue"`
    Profit        float64 `json:"profit"`
    ProfitMargin  float64 `json:"profit_margin"`

    FreeUsers     int     `json:"free_users"`
    PaidUsers     int     `json:"paid_users"`
    TotalEvents   int64   `json:"total_events"`
    StorageUsed   float64 `json:"storage_used_gb"`
}

func (s *DashboardService) GetCostMetrics(ctx context.Context) (*CostDashboard, error) {
    dashboard := &CostDashboard{}

    // 获取成本数据
    dashboard.ServerCost = s.getServerCost(ctx)
    dashboard.DatabaseCost = s.getDatabaseCost(ctx)
    dashboard.AICost = s.getAICost(ctx)
    dashboard.EmailCost = s.getEmailCost(ctx)
    dashboard.TotalCost = dashboard.ServerCost + dashboard.DatabaseCost +
                          dashboard.AICost + dashboard.EmailCost

    // 获取收入数据
    dashboard.Revenue = s.getRevenue(ctx)
    dashboard.Profit = dashboard.Revenue - dashboard.TotalCost
    dashboard.ProfitMargin = dashboard.Profit / dashboard.Revenue * 100

    // 获取用户数据
    dashboard.FreeUsers, _ = s.repo.CountUsers(ctx, "free")
    dashboard.PaidUsers, _ = s.repo.CountUsers(ctx, "pro")

    // 获取存储数据
    dashboard.TotalEvents, _ = s.repo.CountTotalEvents(ctx)
    dashboard.StorageUsed, _ = s.getStorageUsed(ctx)

    return dashboard, nil
}
```

```go
        },
        Message: "Database storage exceeded 8GB",
    },

        Message: "Free users ratio > 90%",
    },
}

    if err != nil {
        return
    }

    for _, rule := range alertRules {
        if rule.Condition(dashboard) {
            s.sendAlert(rule.Name, rule.Message, dashboard)
        }
    }
```
