### 核心代码实现

#### 1. 主入口 (cmd/server/main.go)

```go
package main

import (
    "context"
    "log"
    "net/http"
    "os"
    "os/signal"
    "syscall"
    "time"

    "github.com/gin-gonic/gin"
    "simpletrack/internal/api"
    "simpletrack/internal/config"
    "simpletrack/internal/repository"
    "simpletrack/internal/service"
)

func main() {
    // Load configuration
    cfg, err := config.Load()
    if err != nil {
        log.Fatal("Failed to load config:", err)
    }

    // Initialize database
    db, err := repository.NewPostgresDB(cfg.DatabaseURL)
    if err != nil {
        log.Fatal("Failed to connect to database:", err)
    }
    defer db.Close()

    // Initialize Redis
    redis, err := repository.NewRedisClient(cfg.RedisURL)
    if err != nil {
        log.Fatal("Failed to connect to Redis:", err)
    }
    defer redis.Close()

    // Initialize repositories
    repos := repository.NewRepositories(db)

    // Initialize services
    services := service.NewServices(repos, redis, cfg)

    // Initialize API handlers
    handlers := api.NewHandler(services, cfg)

    // Setup Gin router
    router := setupRouter(handlers, cfg)

    // Start server
    srv := &http.Server{
        Addr:    ":" + cfg.Port,
        Handler: router,
    }

    // Graceful shutdown
    go func() {
        if err := srv.ListenAndServe(); err != nil && err != http.ErrServerClosed {
            log.Fatal("Server failed:", err)
        }
    }()

    log.Printf("Server started on port %s", cfg.Port)

    // Wait for interrupt signal
    quit := make(chan os.Signal, 1)
    signal.Notify(quit, syscall.SIGINT, syscall.SIGTERM)
    <-quit

    log.Println("Shutting down server...")

    ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
    defer cancel()

    if err := srv.Shutdown(ctx); err != nil {
        log.Fatal("Server forced to shutdown:", err)
    }

    log.Println("Server exited")
}

func setupRouter(h *api.Handler, cfg *config.Config) *gin.Engine {
    if cfg.Environment == "production" {
        gin.SetMode(gin.ReleaseMode)
    }

    r := gin.Default()

    // Middleware
    r.Use(api.CORSMiddleware())
    r.Use(api.LoggerMiddleware())

    // Public routes
    public := r.Group("/api/v1")
    {
        // Event collection (no auth required)
        public.POST("/events", h.CreateEvent)

        // Auth
        public.POST("/auth/register", h.Register)
        public.POST("/auth/login", h.Login)
    }

    // Protected routes
    protected := r.Group("/api/v1")
    protected.Use(api.AuthMiddleware(cfg.JWTSecret))
    {
        // Sites
        protected.GET("/sites", h.ListSites)
        protected.POST("/sites", h.CreateSite)
        protected.GET("/sites/:id", h.GetSite)
        protected.PUT("/sites/:id", h.UpdateSite)
        protected.DELETE("/sites/:id", h.DeleteSite)

        // Stats
        protected.GET("/sites/:id/stats", h.GetStats)
        protected.POST("/sites/:id/funnels/analyze", h.AnalyzeFunnel)
        protected.GET("/sites/:id/retention", h.GetRetention)

        // AI Insights
        protected.GET("/sites/:id/insights", h.GetInsights)
        protected.POST("/sites/:id/insights/generate", h.GenerateInsight)
    }

    // Health check
    r.GET("/health", func(c *gin.Context) {
        c.JSON(200, gin.H{"status": "ok"})
    })

    return r
}
```

```go
package api

import (
    "net/http"
    "time"

    "github.com/gin-gonic/gin"
    "simpletrack/internal/models"
)

type EventRequest struct {
    Site       string                 `json:"site" binding:"required"`
    Event      string                 `json:"event" binding:"required,max=100"`
    UserID     string                 `json:"user_id" binding:"max=255"`
    SessionID  string                 `json:"session_id" binding:"max=255"`
    Properties map[string]interface{} `json:"properties"`
    Timestamp  int64                  `json:"timestamp"`
    URL        string                 `json:"url" binding:"max=2048"`
    Referrer   string                 `json:"referrer" binding:"max=2048"`
}

func (h *Handler) CreateEvent(c *gin.Context) {
    var req EventRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }

    // Verify API key and get site
    site, err := h.services.Site.GetByAPIKey(c.Request.Context(), req.Site)
    if err != nil {
        c.JSON(http.StatusUnauthorized, gin.H{"error": "Invalid API key"})
        return
    }

    // Check rate limit based on plan
    if err := h.services.Event.CheckRateLimit(c.Request.Context(), site); err != nil {
        c.JSON(http.StatusTooManyRequests, gin.H{"error": "Rate limit exceeded"})
        return
    }

    // Parse IP and User Agent
    ip := c.ClientIP()
    userAgent := c.Request.UserAgent()

    // Get country from IP (using GeoIP or similar)
    country := h.services.Event.GetCountryFromIP(ip)

    // Create event
    event := &models.Event{
        SiteID:     site.ID,
        EventName:  req.Event,
        UserID:     req.UserID,
        SessionID:  req.SessionID,
        Properties: req.Properties,
        URL:        req.URL,
        Referrer:   req.Referrer,
        UserAgent:  userAgent,
        IPAddress:  ip,
        Country:    country,
        CreatedAt:  time.Now(),
    }

    if req.Timestamp > 0 {
        event.CreatedAt = time.Unix(req.Timestamp/1000, 0)
    }

    // Save event (async via queue for better performance)
    if err := h.services.Event.CreateAsync(c.Request.Context(), event); err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": "Failed to save event"})
        return
    }

    // Return 204 No Content for fast response
    c.Status(http.StatusNoContent)
}
```

#### 3. 漏斗分析服务 (internal/service/funnel.go)

```go
package service

import (
    "context"
    "time"

    "simpletrack/internal/models"
    "simpletrack/internal/repository"
)

type FunnelService struct {
    repo  *repository.Repositories
    cache *repository.RedisClient
}

type FunnelStep struct {
    Event      string  `json:"event"`
    Count      int     `json:"count"`
    Conversion float64 `json:"conversion"`
    DropOff    int     `json:"drop_off"`
}

type FunnelResult struct {
    Steps      []FunnelStep `json:"steps"`
    TotalUsers int          `json:"total_users"`
    StartDate  time.Time    `json:"start_date"`
    EndDate    time.Time    `json:"end_date"`
}

func (s *FunnelService) Analyze(ctx context.Context, siteID int, events []string, from, to time.Time) (*FunnelResult, error) {
    // Check cache first
    cacheKey := fmt.Sprintf("funnel:%d:%v:%s:%s", siteID, events, from.Format("2006-01-02"), to.Format("2006-01-02"))
    if cached, err := s.cache.Get(ctx, cacheKey); err == nil {
        var result FunnelResult
        if err := json.Unmarshal([]byte(cached), &result); err == nil {
            return &result, nil
        }
    }

    // Calculate funnel
    steps := make([]FunnelStep, len(events))
    totalUsers := 0

    for i, eventName := range events {
        var count int
        var err error

        if i == 0 {
            // First step: count all unique users
            count, err = s.repo.Event.CountUniqueUsers(ctx, siteID, eventName, from, to)
            totalUsers = count
        } else {
            // Subsequent steps: count users who completed previous step
            prevEvent := events[i-1]
            count, err = s.repo.Event.CountUsersCompletedBothEvents(ctx, siteID, prevEvent, eventName, from, to)
        }

        if err != nil {
            return nil, err
        }

        conversion := 0.0
        if totalUsers > 0 {
            conversion = float64(count) / float64(totalUsers) * 100
        }

        dropOff := 0
        if i > 0 {
            dropOff = steps[i-1].Count - count
        }

        steps[i] = FunnelStep{
            Event:      eventName,
            Count:      count,
            Conversion: conversion,
            DropOff:    dropOff,
        }
    }

    result := &FunnelResult{
        Steps:      steps,
        TotalUsers: totalUsers,
        StartDate:  from,
        EndDate:    to,
    }

    // Cache result for 1 hour
    if data, err := json.Marshal(result); err == nil {
        s.cache.Set(ctx, cacheKey, string(data), time.Hour)
    }

    return result, nil
}
```

```go
// 多层缓存
type CacheService struct {
    redis  *redis.Client
    local  *cache.Cache // 本地内存缓存
}

func (s *CacheService) Get(ctx context.Context, key string) (interface{}, error) {
    // 1. 先查本地缓存（最快）
    if val, found := s.local.Get(key); found {
        return val, nil
    }

    // 2. 查Redis缓存
    val, err := s.redis.Get(ctx, key).Result()
    if err == nil {
        // 写入本地缓存
        s.local.Set(key, val, 5*time.Minute)
        return val, nil
    }

    return nil, err
}
```

```sql
-- 按月分区事件表
CREATE TABLE events (
    id BIGSERIAL,
    site_id INT NOT NULL,
    event_name VARCHAR(100) NOT NULL,
    user_id VARCHAR(255),
    session_id VARCHAR(255),
    properties JSONB,
    url TEXT,
    referrer TEXT,
    user_agent TEXT,
    ip_address INET,
    country VARCHAR(2),
    created_at TIMESTAMP NOT NULL DEFAULT NOW(),
    PRIMARY KEY (id, created_at)
) PARTITION BY RANGE (created_at);

-- 创建分区
CREATE TABLE events_2026_01 PARTITION OF events
    FOR VALUES FROM ('2026-01-01') TO ('2026-02-01');

CREATE TABLE events_2026_02 PARTITION OF events
    FOR VALUES FROM ('2026-02-01') TO ('2026-03-01');

-- 自动创建分区的函数
CREATE OR REPLACE FUNCTION create_monthly_partition()
RETURNS void AS $$
DECLARE
    partition_date DATE;
    partition_name TEXT;
    start_date TEXT;
    end_date TEXT;
BEGIN
    partition_date := DATE_TRUNC('month', NOW() + INTERVAL '1 month');
    partition_name := 'events_' || TO_CHAR(partition_date, 'YYYY_MM');
    start_date := partition_date::TEXT;
    end_date := (partition_date + INTERVAL '1 month')::TEXT;

    EXECUTE format('CREATE TABLE IF NOT EXISTS %I PARTITION OF events FOR VALUES FROM (%L) TO (%L)',
        partition_name, start_date, end_date);
END;
$$ LANGUAGE plpgsql;

-- 定时任务：每月自动创建下个月的分区
SELECT cron.schedule('create-partition', '0 0 1 * *', 'SELECT create_monthly_partition()');
```

```sql
CREATE MATERIALIZED VIEW daily_stats AS
SELECT
    site_id,
    DATE(created_at) as date,
    COUNT(*) FILTER (WHERE event_name = 'pageview') as pageviews,
    COUNT(DISTINCT user_id) as unique_visitors,
    COUNT(DISTINCT session_id) as sessions,
    COUNT(*) FILTER (WHERE event_name = 'signup') as signups,
    COUNT(*) FILTER (WHERE event_name = 'payment') as payments
FROM events
GROUP BY site_id, DATE(created_at);

-- 创建索引
CREATE UNIQUE INDEX ON daily_stats(site_id, date);

-- 定时刷新（每小时）
CREATE OR REPLACE FUNCTION refresh_daily_stats()
RETURNS void AS $$
BEGIN
    REFRESH MATERIALIZED VIEW CONCURRENTLY daily_stats;
END;
$$ LANGUAGE plpgsql;

SELECT cron.schedule('refresh-stats', '0 * * * *', 'SELECT refresh_daily_stats()');
```

```go
        openaiClient: openai.NewClient(apiKey),
        cache:      cache,

// GenerateWeeklyInsight 生成周报洞察
func (s *AIService) GenerateWeeklyInsight(ctx context.Context, siteID int) (*models.Insight, error) {
    // 1. 收集数据
    endDate := time.Now()
    startDate := endDate.AddDate(0, 0, -7)

    stats, err := s.collectWeeklyStats(ctx, siteID, startDate, endDate)
    if err != nil {
        return nil, fmt.Errorf("failed to collect stats: %w", err)
    }

    // 2. 构建提示词
    prompt := s.buildInsightPrompt(stats)

    // 3. 调用OpenAI API
    content, err := s.callOpenAI(ctx, prompt)
    if err != nil {
        return nil, fmt.Errorf("failed to call OpenAI: %w", err)
    }

    // 4. 保存洞察
    insight := &models.Insight{
        SiteID:      siteID,
        PeriodStart: startDate,
        PeriodEnd:   endDate,
        Content:     content,
        Metadata:    stats,
        CreatedAt:   time.Now(),
    }

    if err := s.repo.Insight.Create(ctx, insight); err != nil {
        return nil, fmt.Errorf("failed to save insight: %w", err)
    }

    return insight, nil
}

// collectWeeklyStats 收集一周的统计数据
func (s *AIService) collectWeeklyStats(ctx context.Context, siteID int, start, end time.Time) (map[string]interface{}, error) {
    // 当前周数据
    currentStats, err := s.repo.Event.GetPeriodStats(ctx, siteID, start, end)
    if err != nil {
        return nil, err
    }

    // 上周数据（用于对比）
    prevStart := start.AddDate(0, 0, -7)
    prevEnd := end.AddDate(0, 0, -7)
    prevStats, err := s.repo.Event.GetPeriodStats(ctx, siteID, prevStart, prevEnd)
    if err != nil {
        return nil, err
    }

    // 计算变化率
    stats := map[string]interface{}{
        "current": currentStats,
        "previous": prevStats,
        "changes": map[string]float64{
            "pageviews":    calculateChange(currentStats.Pageviews, prevStats.Pageviews),
            "visitors":     calculateChange(currentStats.Visitors, prevStats.Visitors),
            "signups":      calculateChange(currentStats.Signups, prevStats.Signups),
            "payments":     calculateChange(currentStats.Payments, prevStats.Payments),
            "signup_rate":  currentStats.SignupRate - prevStats.SignupRate,
            "payment_rate": currentStats.PaymentRate - prevStats.PaymentRate,
        },
    }

    // 获取热门页面
    topPages, _ := s.repo.Event.GetTopPages(ctx, siteID, start, end, 5)
    stats["top_pages"] = topPages

    // 获取流量来源
    sources, _ := s.repo.Event.GetTrafficSources(ctx, siteID, start, end)
    stats["sources"] = sources

    return stats, nil
}

// callOpenAI 调用OpenAI API
func (s *AIService) callOpenAI(ctx context.Context, prompt string) (string, error) {
    resp, err := s.openaiClient.CreateChatCompletion(
        ctx,
        openai.ChatCompletionRequest{
            Model: openai.GPT4TurboPreview, // 或 GPT4o
            Messages: []openai.ChatCompletionMessage{
                {
                    Role:    openai.ChatMessageRoleSystem,
                    Content: "你是一个专业的SaaS数据分析师，擅长发现数据中的问题并给出可操作的建议。",
                },
                {
                    Role:    openai.ChatMessageRoleUser,
                    Content: prompt,
                },
            },
            Temperature: 0.7,
            MaxTokens:   1000,
        },
    )

    if err != nil {
        return "", err
    }

    if len(resp.Choices) == 0 {
        return "", fmt.Errorf("no response from OpenAI")
    }

    return resp.Choices[0].Message.Content, nil
}

// calculateChange 计算变化率
func calculateChange(current, previous int) float64 {
    if previous == 0 {
        if current > 0 {
            return 100.0
        }
        return 0.0
    }
    return float64(current-previous) / float64(previous) * 100
}
```

```go
// DetectAnomalies 检测异常
func (s *AIService) DetectAnomalies(ctx context.Context, siteID int) ([]models.Anomaly, error) {
    // 获取最近30天的每日数据
    endDate := time.Now()
    startDate := endDate.AddDate(0, 0, -30)

    dailyStats, err := s.repo.Event.GetDailyStats(ctx, siteID, startDate, endDate)
    if err != nil {
        return nil, err
    }

    var anomalies []models.Anomaly

    // 计算平均值和标准差
    avgPageviews, stdPageviews := calculateStats(dailyStats, "pageviews")
    avgSignupRate, stdSignupRate := calculateStats(dailyStats, "signup_rate")

    // 检测今日数据是否异常（超过2个标准差）
    today := dailyStats[len(dailyStats)-1]

    // 页面浏览量异常
    if math.Abs(float64(today.Pageviews)-avgPageviews) > 2*stdPageviews {
        anomaly := models.Anomaly{
            Type:        "pageviews",
            Current:     today.Pageviews,
            Expected:    int(avgPageviews),
            Severity:    calculateSeverity(today.Pageviews, avgPageviews, stdPageviews),
            Description: fmt.Sprintf("页面浏览量异常：当前%d，预期%d", today.Pageviews, int(avgPageviews)),
        }
        anomalies = append(anomalies, anomaly)
    }

    // 注册转化率异常
    if math.Abs(today.SignupRate-avgSignupRate) > 2*stdSignupRate {
        anomaly := models.Anomaly{
            Type:        "signup_rate",
            Current:     int(today.SignupRate * 100),
            Expected:    int(avgSignupRate * 100),
            Severity:    calculateSeverity(int(today.SignupRate*100), avgSignupRate*100, stdSignupRate*100),
            Description: fmt.Sprintf("注册转化率异常：当前%.2f%%，预期%.2f%%", today.SignupRate, avgSignupRate),
        }
        anomalies = append(anomalies, anomaly)
    }

    return anomalies, nil
}
```

```go
    // 每小时检测异常
    c.AddFunc("0 * * * *", func() {
        log.Println("Starting anomaly detection...")
        detectAnomalies(context.Background(), services)
    })

    // 每天凌晨3点刷新物化视图
    c.AddFunc("0 3 * * *", func() {
        log.Println("Refreshing materialized views...")
        refreshMaterializedViews(context.Background(), repos)
    })

    c.Start()
    log.Println("Worker started")

    // 保持运行
    select {}
}

func generateWeeklyInsights(ctx context.Context, services *service.Services) {
    // 获取所有付费用户的网站
    sites, err := services.Site.GetAllPaidSites(ctx)
    if err != nil {
        log.Printf("Failed to get paid sites: %v", err)
        return
    }

    for _, site := range sites {
        insight, err := services.AI.GenerateWeeklyInsight(ctx, site.ID)
        if err != nil {
            log.Printf("Failed to generate insight for site %d: %v", site.ID, err)
            continue
        }

        // 发送邮件通知
        if err := services.Email.SendInsightEmail(ctx, site, insight); err != nil {
            log.Printf("Failed to send email for site %d: %v", site.ID, err)
        }

        log.Printf("Generated insight for site %d", site.ID)
    }
}
```

```go
// internal/middleware/metrics.go
package middleware

import (
    "time"

    "github.com/gin-gonic/gin"
    "github.com/prometheus/client_golang/prometheus"
    "github.com/prometheus/client_golang/prometheus/promauto"
)

var (
    httpRequestsTotal = promauto.NewCounterVec(
        prometheus.CounterOpts{
            Name: "http_requests_total",
            Help: "Total number of HTTP requests",
        },
        []string{"method", "endpoint", "status"},
    )

    httpRequestDuration = promauto.NewHistogramVec(
        prometheus.HistogramOpts{
            Name:    "http_request_duration_seconds",
            Help:    "HTTP request duration in seconds",
            Buckets: prometheus.DefBuckets,
        },
        []string{"method", "endpoint"},
    )
)

func PrometheusMiddleware() gin.HandlerFunc {
    return func(c *gin.Context) {
        start := time.Now()

        c.Next()

        duration := time.Since(start).Seconds()
        status := c.Writer.Status()

        httpRequestsTotal.WithLabelValues(
            c.Request.Method,
            c.FullPath(),
            fmt.Sprintf("%d", status),
        ).Inc()

        httpRequestDuration.WithLabelValues(
            c.Request.Method,
            c.FullPath(),
        ).Observe(duration)
    }
}
```

```bash
# 删除30天前的备份

  awk '{print $4}' | \
  while read file; do
    if [[ $(date -d "$(echo $file | cut -d_ -f2 | cut -d. -f1)" +%s) -lt $(date -d "30 days ago" +%s) ]]; then

    fi
  done
```
