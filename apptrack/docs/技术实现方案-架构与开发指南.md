```swift
// AppTrack.swift
public class AppTrack {
    public static let shared = AppTrack()
    private var apiKey: String?
    private let tracker: EventTracker
    private let autoTracker: AutoTracker

    private init() {
        self.tracker = EventTracker()
        self.autoTracker = AutoTracker()
    }

    // Initialize SDK
    public static func initialize(apiKey: String) {
        shared.apiKey = apiKey
        shared.autoTracker.start()
    }

    // Track custom event
    public static func track(_ eventName: String,
                            properties: [String: Any]? = nil) {
        shared.tracker.track(eventName, properties: properties)
    }
}

// EventTracker.swift
class EventTracker {
    private let networkManager = NetworkManager()
    private let storageManager = StorageManager()
    private var eventQueue: [Event] = []

    func track(_ eventName: String, properties: [String: Any]?) {
        let event = Event(
            name: eventName,
            properties: properties,
            timestamp: Date(),
            userId: getUserId(),
            sessionId: getSessionId()
        )

        eventQueue.append(event)

        // Batch send when queue reaches 10 events
        if eventQueue.count >= 10 {
            flush()
        }
    }

    func flush() {
        guard !eventQueue.isEmpty else { return }

        networkManager.sendEvents(eventQueue) { [weak self] result in
            switch result {
            case .success:
                self?.eventQueue.removeAll()
            case .failure:
                // Save to local storage for retry
                self?.storageManager.save(self?.eventQueue ?? [])
            }
        }
    }
}

// AutoTracker.swift
class AutoTracker {
    func start() {
        // Track app lifecycle
        NotificationCenter.default.addObserver(
            self,
            selector: #selector(appDidBecomeActive),
            name: UIApplication.didBecomeActiveNotification,
            object: nil
        )

        NotificationCenter.default.addObserver(
            self,
            selector: #selector(appWillResignActive),
            name: UIApplication.willResignActiveNotification,
            object: nil
        )

        // Swizzle UIViewController for screen tracking
        swizzleViewDidAppear()
    }

    @objc private func appDidBecomeActive() {
        AppTrack.track("app_opened")
    }

    @objc private func appWillResignActive() {
        AppTrack.track("app_closed")
    }

    private func swizzleViewDidAppear() {
        // Method swizzling to auto-track screen views
        let originalSelector = #selector(UIViewController.viewDidAppear(_:))
        let swizzledSelector = #selector(UIViewController.at_viewDidAppear(_:))

        guard let originalMethod = class_getInstanceMethod(
            UIViewController.self, originalSelector),
              let swizzledMethod = class_getInstanceMethod(
            UIViewController.self, swizzledSelector)
        else { return }

        method_exchangeImplementations(originalMethod, swizzledMethod)
    }
}

extension UIViewController {
    @objc func at_viewDidAppear(_ animated: Bool) {
        self.at_viewDidAppear(animated)

        let screenName = String(describing: type(of: self))
        AppTrack.track("screen_view", properties: [
            "screen_name": screenName
        ])
    }
}
```

```kotlin
object AppTrack {
    private var apiKey: String? = null
    private lateinit var context: Context
    private val tracker = EventTracker()
    private val autoTracker = AutoTracker()

    fun initialize(context: Context, apiKey: String) {
        this.context = context.applicationContext
        this.apiKey = apiKey
        autoTracker.start(context)
    }

    fun track(eventName: String, properties: Map<String, Any>? = null) {
        tracker.track(eventName, properties)
    }
}

// EventTracker.kt
class EventTracker {
    private val networkManager = NetworkManager()
    private val storageManager = StorageManager()
    private val eventQueue = mutableListOf<Event>()

    fun track(eventName: String, properties: Map<String, Any>?) {
        val event = Event(
            name = eventName,
            properties = properties,
            timestamp = System.currentTimeMillis(),
            userId = getUserId(),
            sessionId = getSessionId()
        )

        eventQueue.add(event)

        if (eventQueue.size >= 10) {
            flush()
        }
    }

    fun flush() {
        if (eventQueue.isEmpty()) return

        networkManager.sendEvents(eventQueue) { result ->
            when (result) {
                is Result.Success -> eventQueue.clear()
                is Result.Failure -> storageManager.save(eventQueue)
            }
        }
    }
}

// AutoTracker.kt
class AutoTracker {
    fun start(context: Context) {

                    AppTrack.track("screen_view", mapOf(
                        "screen_name" to activity.javaClass.simpleName
                    ))
                }

                override fun onActivityStarted(activity: Activity) {
                    AppTrack.track("app_opened")
                }

                override fun onActivityStopped(activity: Activity) {
                    AppTrack.track("app_closed")
                }

                // Other lifecycle methods...
            }
        )
    }
}
```

### 核心实现

**事件接收API**：

```go
// handlers/events.go
package handlers

type EventHandler struct {
    service *service.EventService
    limiter *middleware.RateLimiter
}

func (h *EventHandler) HandleEvent(c *gin.Context) {
    var req struct {
        Events []models.Event `json:"events"`
    }

    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(400, gin.H{"error": "Invalid request"})
        return
    }

    // Rate limiting
    if err := h.limiter.Check(c); err != nil {
        c.JSON(429, gin.H{"error": "Too many requests"})
        return
    }

    // Process events
    if err := h.service.ProcessEvents(c.Request.Context(), req.Events); err != nil {
        c.JSON(500, gin.H{"error": "Failed to process events"})
        return
    }

    c.JSON(200, gin.H{"status": "ok"})
}

// service/event_service.go
package service

type EventService struct {
    repo  *repository.EventRepository
    cache *redis.Client
}

func (s *EventService) ProcessEvents(ctx context.Context, events []models.Event) error {
    // Validate events
    for _, event := range events {
        if err := s.validateEvent(event); err != nil {
            return err
        }
    }

    // Batch insert to ClickHouse
    if err := s.repo.BatchInsert(ctx, events); err != nil {
        return err
    }

    // Update cache
    s.updateCache(ctx, events)

    return nil
}
```

```go
// handlers/analytics.go
package handlers

type AnalyticsHandler struct {
    service *service.AnalyticsService
}

func (h *AnalyticsHandler) GetRetention(c *gin.Context) {
    appID := c.Param("appId")

    retention, err := h.service.GetRetention(c.Request.Context(), appID)
    if err != nil {
        c.JSON(500, gin.H{"error": err.Error()})
        return
    }

    c.JSON(200, retention)
}

// service/analytics_service.go
package service

type AnalyticsService struct {
    repo  *repository.EventRepository
    cache *redis.Client
}

func (s *AnalyticsService) GetRetention(ctx context.Context, appID string) (*models.Retention, error) {
    // Check cache
    cacheKey := fmt.Sprintf("retention:%s", appID)
    if cached, err := s.cache.Get(ctx, cacheKey).Result(); err == nil {
        var retention models.Retention
        json.Unmarshal([]byte(cached), &retention)
        return &retention, nil
    }

    // Query ClickHouse
    retention, err := s.repo.CalculateRetention(ctx, appID)
    if err != nil {
        return nil, err
    }

    // Cache result
    data, _ := json.Marshal(retention)
    s.cache.Set(ctx, cacheKey, data, 5*time.Minute)

    return retention, nil
}
```

### ClickHouse Schema

```sql
-- Events table
CREATE TABLE events (
    event_id UUID DEFAULT generateUUIDv4(),
    app_id String,
    user_id String,
    session_id String,
    event_name String,
    properties String, -- JSON
    timestamp DateTime64(3),
    device_type String,
    os_name String,
    os_version String,
    app_version String,
    country String,
    city String
) ENGINE = MergeTree()
PARTITION BY toYYYYMM(timestamp)
ORDER BY (app_id, timestamp, user_id)
TTL timestamp + INTERVAL 90 DAY;

-- Materialized view for daily stats
CREATE MATERIALIZED VIEW daily_stats
ENGINE = SummingMergeTree()
PARTITION BY toYYYYMM(date)
ORDER BY (app_id, date)
AS SELECT
    app_id,
    toDate(timestamp) as date,
    count() as event_count,
    uniq(user_id) as unique_users,
    uniq(session_id) as sessions
FROM events
GROUP BY app_id, date;

-- Retention calculation
CREATE MATERIALIZED VIEW retention_cohorts
ENGINE = AggregatingMergeTree()
PARTITION BY toYYYYMM(cohort_date)
ORDER BY (app_id, cohort_date, day_number)
AS SELECT
    app_id,
    toDate(min(timestamp)) as cohort_date,
    user_id,
    dateDiff('day', cohort_date, toDate(timestamp)) as day_number,
    uniqState(user_id) as users
FROM events
GROUP BY app_id, user_id, cohort_date, day_number;
```

```go
// service/ai_service.go
package service

type AIService struct {
    openai *openai.Client
    repo   *repository.EventRepository
}

func (s *AIService) GenerateWeeklyInsight(ctx context.Context, appID string) (*models.Insight, error) {
    // 1. Gather data
    data, err := s.gatherData(ctx, appID)
    if err != nil {
        return nil, err
    }

    // 2. Build prompt
    prompt := s.buildPrompt(data)

    // 3. Call OpenAI
    resp, err := s.openai.CreateChatCompletion(ctx, openai.ChatCompletionRequest{
        Model: openai.GPT4TurboPreview,
        Messages: []openai.ChatCompletionMessage{
            {
                Role:    openai.ChatMessageRoleSystem,
                Content: "You are a mobile app analytics expert.",
            },
            {
                Role:    openai.ChatMessageRoleUser,
                Content: prompt,
            },
        },
        Temperature: 0.7,
        MaxTokens:   1000,
    })

    if err != nil {
        return nil, err
    }

    // 4. Parse response
    insight := s.parseInsight(resp.Choices[0].Message.Content)

    return insight, nil
}

%s

Provide:
1. Key findings (2-3 points)

Keep it concise and actionable.
```

## 💡 关键建议

### 性能优化

1. **批量处理**：事件批量写入ClickHouse
2. **缓存策略**：Redis缓存热点数据
3. **异步处理**：AI分析异步执行
4. **数据分区**：按月分区，自动清理

### 安全措施

1. **API认证**：JWT token
2. **限流保护**：IP + 用户双重限流
3. **数据加密**：传输加密（HTTPS）
4. **SQL注入防护**：参数化查询

### 可扩展性

1. **水平扩展**：无状态设计
2. **数据库分片**：按app_id分片
